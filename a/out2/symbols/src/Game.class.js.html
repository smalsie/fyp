1 /**
  2 * Game class which is at the core of the project
  3 *
  4 * @class Game
  5 * @constructor
  6 * @param {int} width : Width of the game
  7 * @param {int} height : Height of the game
  8 * @param {String} name : The name of the game
  9 */
 10 function Game(width, height,  name){
 11 
 12     document.title = name
 13     
 14     if (typeof phaserType === 'undefined') { phaserType = Phaser.AUTO; }
 15     
 16     this.world = new Phaser.Game(width, height, Phaser.AUTO, '', { preload: preload, create: create, update: update });
 17     
 18     /**
 19 	* Game Height function that returns the height of the game, useful when 
 20 	* a percentage is give as the size in pixels will be returned
 21 	*
 22 	* @method gameHeight
 23 	* @return {Int} Returns the height of the game in pixels
 24 	*/
 25     this.gameHeight = function(){
 26         return this.world.world.height;
 27     };
 28     
 29     /**
 30 	* Game Width function that returns the height of the game, useful when 
 31 	* a percentage is give as the size in pixels will be returned
 32 	*
 33 	* @method gameWidth
 34 	* @return {Int} Returns the width of the game in pixels
 35 	*/   
 36     this.gameWidth = function(){
 37         return this.world.world.width;
 38     };
 39    
 40     
 41 };
